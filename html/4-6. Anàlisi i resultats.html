def correlations(atributs, by):
    sns.set(style="whitegrid")
    #Prepare the data    
    cleaned = pd.read_csv("cleaned_train.csv")
    cleaned[["SibSp","Parch","Pclass","Survived"]] = cleaned[["SibSp","Parch","Pclass","Survived"]].astype(str)
    
    #Get dummies and set 'by' correlation
    at = atributs + [by]
    dummies = pd.get_dummies(cleaned[at])
    
    #Correlation matrix
    correlations = dummies.corr()
    
    #Prepare to plot
    l = len(pd.get_dummies(cleaned[by]).columns)
    fig, ax = plt.subplots(l, figsize=(20,8))
    plt.subplots_adjust(top=2)
    i=0
    palette=["orange","blue","red","grey","green","yellow"]
    for b in pd.get_dummies(cleaned[by]):
        st = by+"_"+b
        corr_s = correlations[st]
        corr_s = corr_s.sort_values(ascending=False)
        #Seaborn plot
        ax[i].set_title(st)
        
        sns.barplot(corr_s[1:-1],corr_s.index[1:-1], ax=ax[i], palette=[palette[i%6]])
        i+=1

#Per calcular la precisi√≥ no n'hi ha prou amb comparar els valors. Es provar√† amb marges d'error de +-5% incrementanls:

    count=0
    for real,predicted in zip(test_age,p_age):
